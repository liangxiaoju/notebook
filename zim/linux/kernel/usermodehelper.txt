static void exec_func(struct work_struct *data)
{
	char *filename = "sh";

	char *envp[] = {
		"HOME=/",
		"PATH=/sbin:/bin:/system/bin:/system/xbin",
		NULL,
	};

	char *argv[] = {
		filename,
		"dmesg",
		NULL,
	};

	int ret;

	ret = call_usermodehelper(filename, argv, envp, 1);
}
